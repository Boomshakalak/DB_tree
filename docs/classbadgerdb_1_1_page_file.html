    std::ios_base::openmode mode =

    const bool already_exists = exists(filename_);
    if (create_new) {
      // Error if we try to overwrite an existing file.
      if (already_exists) {

      }
      // New files have to be truncated on open.
      mode = mode | std::fstream::trunc;
